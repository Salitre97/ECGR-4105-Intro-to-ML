def train_model(n_epochs, optimizer, model, loss_fn, x_train, x_val, y_train, y_val):
    train_losses = []
    val_losses = []
    
    for epoch in range(1, n_epochs + 1):
        model.train()
        optimizer.zero_grad()
        outputs = model(x_train)
        loss = loss_fn(outputs, y_train)
        loss.backward()
        optimizer.step()                

        #train_preds = outputs.argmax(dim=1)

        model.eval()
        with torch.no_grad():
            val_outputs = model(x_val)
            val_loss = loss_fn(val_outputs, y_val)
            #val_preds = val_outputs.argmax(dim=1) 

        # Calculate metrics for training data
        #train_accuracy = accuracy_score(y_train.cpu(), train_preds.cpu())

        # Calculate metrics for validation data
        #val_accuracy = accuracy_score(y_val.cpu(), val_preds.cpu())

        if epoch == 1 or epoch % 500 == 0:
            print(f"Epoch {epoch}/{n_epochs}, Training loss: {loss.item():.4f}, Validation loss: {val_loss.item():.4f}")
            #print(f"Training Accuracy: {train_accuracy:.4f}")
            #print(f"Validation Accuracy: {val_accuracy:.4f}")

        train_losses.append(loss.detach().item())
        val_losses.append(val_loss.detach().item())
    
    return train_losses, val_losses

# Define the model
model3 = nn.Sequential(
    nn.Linear(3 * 32 * 32, 512),
    nn.Tanh(),
    nn.Linear(512, 10),
    nn.LogSoftmax(dim=1)
)
model3.to(device, non_blocking=True)

# Define loss function and optimizer
loss_fn = nn.NLLLoss()
optimizer = optim.SGD(model3.parameters(), lr=1e-2)

# Training loop
n_epochs = 100
start_time = time.time()

all_preds = []
all_labels = []

for epoch in range(n_epochs):
    model3.train()
    total_loss = 0
    for imgs, labels in trainloader:
        imgs, labels = imgs.to(device, non_blocking=True), labels.to(device, non_blocking=True)
        #print("Batch Device:", imgs.device)
        #print("Model device:", next(model3.parameters()).device)
        optimizer.zero_grad()
        outputs = model3(imgs.view(imgs.size(0), -1))
        loss = loss_fn(outputs, labels)
        loss.backward()
        optimizer.step()

        total_loss += loss.item()

    avg_loss = total_loss / len(trainloader)
    print(f"Epoch: {epoch}/{n_epochs}, Average Loss: {avg_loss}")

training_time = time.time() - start_time
print(f"Training completed in {training_time:.2f} seconds.")

# Evaluate the network
correct = 0
total = 0
model3.eval()
with torch.no_grad():
    for imgs, labels in testloader:
        imgs, labels = imgs.to(device, non_blocking=True), labels.to(device, non_blocking=True)
        outputs = model3(imgs.view(imgs.size(0), -1))
        _, predicted = torch.max(outputs, 1)
        total += labels.size(0)
        correct += (predicted == labels).sum().item()

        # Store predictions and actual labels
        all_preds.extend(predicted.view(-1).tolist())
        all_labels.extend(labels.view(-1).tolist())

accuracy = 100 * correct / total
print(f'Accuracy of the network on the test images: {accuracy:.2f} %')

# Calculate the confusion matrix
conf_matrix = confusion_matrix(all_labels, all_preds)

# Calculate the F1 score
f1Score = f1_score(all_labels, all_preds, average='weighted')

#print("Confusion Matrix:\n", conf_matrix)
print("F1 Score: ", f1Score)
